import telebot
from openai import OpenAI
import requests
import random

TELEGRAM_TOKEN = "DEIN_TELEGRAM_TOKEN"
OPENAI_API_KEY = "DEIN_OPENAI_API_KEY"

bot = telebot.TeleBot(TELEGRAM_TOKEN)
client = OpenAI(api_key=OPENAI_API_KEY)

@bot.message_handler(commands=['start'])
def start(message):
bot.reply_to(message, "🕌 Assalamu Alaikum! Ich bin euer Bot 🤖")

@bot.message_handler(commands=['quran'])
def quran(message):
res = requests.get("https://api.alquran.cloud/v1/ayah/random")
data = res.json()
ayah = data['data']['text']
bot.reply_to(message, "📖 " + ayah)

duas = ["اللهم اغفر لي", "اللهم اهدنا", "اللهم ارزقنا"]

@bot.message_handler(commands=['dua'])
def dua(message):
bot.reply_to(message, "🤲 " + random.choice(duas))

@bot.message_handler(func=lambda message: True)
def chat(message):
response = client.chat.completions.create(
model="gpt-4o-mini",
messages=[{"role": "user", "content": message.text}]
)
bot.reply_to(message, response.choices[0].message.content)

bot.polling()
